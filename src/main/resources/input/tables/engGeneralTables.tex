\makeatletter
% \table@toto -> Table toto only without decoration or legends
% \totoonly -> Table only without legends but with \GTdecorate, for
%              the rules (DRM and legends are in the running text)
% \toto -> Table + legends + complex decorate, for the Player's aid

% \table@@toto -> sutables, included via \GTmorecontent, require
%                 special treatment
% \alias@@toto, \title@@toto -> args for the \GT
%         used to make sure the name is the same everywhere.

% creates \totoonly from \table@toto
\newcommand{\make@only}[1]{
  \csdef{#1only}{\csname table@#1\endcsname\GTdecorate}}

% creates \totoonly from \table@@toto, \alias@@toto, \title@@toto
% use \make@@all !
\newcommand{\make@@only}[2]{
  \csdef{#2only}{
    \ifx#1\relax\relax % we should rather test \ifdef \alias@@#2
    \GT{#2}{\csname title@@#2\endcsname}
    \else
    \GT[\csname alias@@#2\endcsname]{#2}{\csname title@@#2\endcsname}
    \fi
    \GTcontent{\csname table@@#2\endcsname}
    \GTdecorate
  } }

% creates \more@@toto to include in \toto
% ##1, ##2 and ##3 are positioning args for \GTmorecontent
% use \make@@all !
\newcommand{\make@@more}[2]{
  \csdef{more@@#2}##1##2##3{
    \GTmorecontent{#2}{##1}{##2}{##3}{\csname table@@#2\endcsname}
    \GTmoretitle{#2}{\csname title@@#2\endcsname}
    \GTdecorate
  }
}
